\begin{block}{Avancement : Versions 0.4.3 $\rightarrow$ 0.5.3}
  \begin{itemize}
  \item Python 2.7 
  \item Intégration de librairies d'extraction de descripteurs audio : Aubio, Yaafe, Vamp (expérimental)
  
  \item Installation par paquet Debian
  \item Documentation :  \url{http://files.parisson.com/timeside/doc/} + doctest
  \item Amélioration du code des grapheurs
  \item Amélioration du code et des fonctionnalités des décodeurs : support de segment d'audio, mise en mémoire de l'audio pour le rejouer \emph{hors-ligne}
  \item Amélioration du code et des fonctionnalités des analyseurs
    \begin{itemize}\footnotesize
      \item Mise en place d'un conteneur pour stocker et sérialiser (\emph{hdf5, yaml, json, xml}) les résultats des analyses et les annotations
      \item Support de différents modes temporels et différents types de résultats
      \item Ajout de la notion d'analyseurs \emph{Parents}
      \item Ajout de méthodes pour générer les graphiques automatiquement (via \emph{matplotlib})
    \end{itemize}
  \item Mise en place d'un outil en ligne de commande : \texttt{timeside-launch}
  \item Décodeur + Encodeur \emph{live} 
  \end{itemize}

\end{block}

  \begin{block}{Installation}
\url{https://github.com/yomguy/TimeSide\#install}
    \begin{itemize}
    \item Installation des dépendances :
\begin{lstlisting}[language=bash, basicstyle=\TINY]
$ echo "deb http://debian.parisson.com/debian/ stable main" |
$ sudo tee -a /etc/apt/sources.list 
$ echo "deb-src http://debian.parisson.com/debian/ stable main" | sudo tee -a /etc/apt/sources.list 
$ sudo apt-get update 
$ sudo apt-get install git 
$ sudo apt-get build-dep python-timeside
\end{lstlisting}

    \item Installation depuis le dépôt \emph{Github} :
\begin{lstlisting}[language=bash, basicstyle=\TINY]
$ git clone https://github.com/yomguy/TimeSide.git 
$ cd TimeSide 
$ git checkout dev 
$ export PYTHONPATH=$PYTHONPATH:`pwd` 
$ python tests/run_all_tests
\end{lstlisting}
\end{itemize}
